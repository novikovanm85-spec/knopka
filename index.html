// ========== ЗВУКИ (длинный громкий сигнал "БИИИИИ", без затухания) ========== let audioCtx = null; function initAudio() { if (!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)(); return audioCtx; } function playBipSound() { try { const ctx = initAudio(); const now = ctx.currentTime; const duration = 4.5; // 4.5 секунды const osc = ctx.createOscillator(); const gain = ctx.createGain(); osc.connect(gain); gain.connect(ctx.destination); osc.frequency.value = 880; // высокая нота — призывный "БИИИИИ" osc.type = 'sine'; // чистый тон, без пердежа // Максимальная громкость, без нарастания и затухания gain.gain.value = 0.6; osc.start(); osc.stop(now + duration); ctx.resume(); } catch(e) { console.log("Sound error"); } } function playStartSound() { playBipSound(); } function playTimeoutSound() { playBipSound(); // тот же самый звук }
